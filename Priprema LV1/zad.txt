#include <stdlib.h>
#include <time.h>
#include <stdio.h>

void gen_arr(float V[], int n, float dg, float gg) {
    srand(time(NULL)); // Postavlja sjeme za generiranje slučajnih brojeva

    for (int i = 0; i < n; i++) {
        V[i] = dg + ((float)rand() / RAND_MAX) * (gg - dg); // Računa vrijednost između dg i gg
    }
}

a) int sekv_pret(float V[], int n, float x ){
    for(int i = 0; i < n; i++){
        if(V[i] == x){
            return i;
        }
    }
    return -1;
}

b)void insertion_sort(float V[], int n){
    int temp, i, j;
    for(i = 1; i < n; i++){
        temp = V[i];
        j = i - 1;
        while (j >= 0 && V[j] > temp){
            V[j + 1] = V[j];
            j = j - 1;
        }
        V[j + 1] = temp;
    }
}

c) int bin_pret(float V[], int n, float x){
    int donjaG = 0;
    int gornjaG = n-1;
    int sredina = 0;
    
    while(donjaG <= gornjaG){
        sredina = (donjaG + gornjaG)/2;
        
        if(V[sredina] > x){
            gornjaG = sredina -1;
        }
        if(V[sredina] < x){
            donjaG = sredina + 1;
        }
        else{
            return sredina;
        }
    }
    if(donjaG > gornjaG){
        return -1;
    }
}
